239  //! Inject a callback to run immediately. Unlike most methods, it
240  //! is safe to call this function from another thread. Being
241  //! thread-safe adds extra overhead, so it does not make sense to
242  //! call this function from the same thread as PollSet::poll. Note
243  //! that \c inject_cb acquires a lock and definitely must <i>not</i>
244  //! be called from a signal handler (or deadlock could ensue).
245  template<typename CB> void inject_cb(CB &&cb) {
246  std::lock_guard<std::mutex> lk(async_cbs_lock_);
247  async_cbs_.emplace_back(std::forward<CB>(cb));
248  if (!async_pending_) {
249  async_pending_ = true;
250  wake();
251  }
252  }

254  //! Execute a task asynchonously in another thread, then run
255  //! callback on the task's result in the main thread. \arg \c work
256  //! is a task to perform asynchronously in another thread, and must
257  //! be convertible to std::function<R()> for some type \c R. \arg
258  //! \c cb is the callback that processes the result in the main
259  //! thread, and must be convertible to std::function<void(R)> for
260  //! the same type \c R.
261  template<typename Work, typename CB> void async(Work &&work, CB &&cb) {
262  using R = decltype(work());
263  async_task<R> *a = new async_task<R> {
264  this, std::forward<Work>(work), std::forward<CB>(cb), nullptr
265  };
266  ++nasync_;
267  std::thread(&async_task<R>::start, a).detach();
268  }
