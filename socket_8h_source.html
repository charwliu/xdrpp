118 //! Self-closing socket. Note that this socket will be closed as soon
119 //! as it goes out of scope, hence it is important to see whether
120 //! functions you pass it to take a "unique_sock &" or a "const
121 //! unique_sock &&". In the latter case, you are expected to keep the
122 //! file descriptor around.
123 class unique_sock {
124  sock_t s_;
125 public:
126  unique_sock() : s_(invalid_sock) {}
127  explicit unique_sock(sock_t s) : s_(s) {}
128  unique_sock(unique_sock &&uf) : s_(uf.release()) {}
129  ~unique_sock() { clear(); }
130  unique_sock &operator=(unique_sock &&uf) {
131  clear();
132  s_ = uf.release();
133  return *this;
134  }
135 
136  //! Return the file descriptor number, but maintain ownership.
137  sock_t get() const { return s_; }
138  //! True if the file descriptor is not -1.
139  explicit operator bool() const { return s_ != invalid_sock; }
140  //! Return the file descriptor number, relinquishing ownership of
141  //! it. The \c unique_sock will have file descriptor -1 after this
142  //! method returns.
143  sock_t release() {
144  sock_t ret = s_;
145  s_ = invalid_sock;
146  return ret;
147  }
148  void clear() {
149  if (s_) {
150  close(s_);
151  s_ = invalid_sock;
152  }
153  }
154  void reset(sock_t s) { clear(); s_ = s; }
155 };
