134 //! Return a std::string containing a pretty-printed version an XDR
135 //! data type. The string will contain multiple lines and end with a
136 //! newline. \arg name if non-NULL, the string begins with the name
137 //! and an equals sign. \arg indent specifies a non-zero minimum
138 //! indentation.
139 template<typename T> std::string
140 xdr_to_string(const T &t, const char *name = nullptr, int indent = 0)
141 {
142  detail::Printer p(indent);
